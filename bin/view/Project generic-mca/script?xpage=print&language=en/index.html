#!/bin/ksh -p

#
# Script to create generic MCA events and to modify existing AMD
# events to the new payload specification.
#
# Extended knowledge article content will be developed after the initial
# seed articles generated in this script.

#
# PSARC case to associate fault events with
#
PSARC="PSARC/2007/XXX"
PSARCDESC="Generic x86 MCA"

#
# Fault dictionary name and id ("GC" in hex)
#
DICTNAME="GCPU"
DICTID="0x4743"
typeset -i msgnum=0

GCPUSUBCLASS="generic-x86"

GERPTSUBCLASS="ereport.cpu.${GCPUSUBCLASS}"
GCPUFLTCLASS="fault.cpu.generic-x86"
GMEMFLTCLASS="fault.memory.generic-x86"

AMDERPTSUBCLASS="ereport.cpu.amd"

#
# Create set
#
erset -c -p description="$PSARCDESC" $PSARC

#
# Create generic x86 ereport subclass
#
erevent -c -p description="Generic x86 MCA Ereports" $GERPTSUBCLASS

#
# Create cpu fault subclass
#
erevent -c -p description="Generic x86 MCA CPU Faults" $GCPUFLTCLASS

#
# Add common payload to the generic x86 ereport subclass for all leaf ereports
# to inherit.
#
erevent -P 0=- $GERPTSUBCLASS < /dev/stderr
		exit 1
	fi

	erevent -c -p description="${a[1]}" "${a[0]}"
	erevent -p leaf=1 "${a[0]}"
	erset -e "${a[0]}" $PSARC

	msgnum=$((msgnum + 1))
	ermsg -c -p keys="${a[0]}" $DICTNAME:$msgnum
	ermsg -P title="${a[2]}" $DICTNAME:$msgnum
	ermsg -p type="Fault" $DICTNAME:$msgnum
	ermsg -p severity="${a[3]}" $DICTNAME:$msgnum
	ermsg -P description="${a[4]}" $DICTNAME:$msgnum
	ermsg -P details="${a[4]}" $DICTNAME:$msgnum
	ermsg -P response="${a[5]}" $DICTNAME:$msgnum
	ermsg -P impact="${a[6]}" $DICTNAME:$msgnum
	ermsg -P action="${a[7]}" $DICTNAME:$msgnum
}

#
# Some common dictionary sentences
#
OFFLINECPU="The system will attempt to offline this cpu to remove it from service."
PERFIMPACT="Performance of this system may be affected."
REPAIR="Schedule a repair procedure to replace the affected CPU.  Use 'fmadm faulty' to identify the module."

function stdcpuflt

	typeset leaf="$1"
	typeset fdesc="$2"
	typeset title="$3"
	typeset desc="$4"

	create_fault <<-EOM
	${GCPUFLTCLASS}.${leaf}
	$fdesc
	$title
	Major
	$desc
	$OFFLINECPU
	$PERFIMPACT
	$REPAIR
EOM
}

#
# I'd have liked to re-use past fault classes, expecting a distinct
# dictionary to give new msgid etc, but that is not the way
# dictionaries work in fmd.  All dictionaries opened by a module
# are noted and when a diagnosis is produced the first dictionary
# in the list that matches on fault class keys contrinutes the
# diag code.  Since other i86pc rules such as amd64.esc may
# load at the same time as us and load different dictionaries
# it's down to luck which one we get.
#

#
# Create generic x86 memory fault subclass
#
erevent -c -p description="Generic x86 Memory Faults" $GMEMFLTCLASS

create_fault <<-EOM
	$GMEMFLTCLASS.page_sb
	Fault in a memory page (internal diagnosis)
	Fault in a memory page (internal diagnosis)
	Minor
	A page of memory is reporting excessive ECC errors.
	The system will attempt to retire the affected page from further use.
	Retiring a single page from use will have negligible performance impact.
	No service or repair action is required;  a separate dimm fault will be diagnosed if appropriate.
EOM

create_fault <<-EOM
	$GMEMFLTCLASS.page_ue
	An uncorrectable error in a memory page (internal diagnosis)
	An uncorrectable error in a memory page (internal diagnosis)
	Major
	An uncorrectable error was reported on a memory page.
	The system will attempt to retire the affected page from further use.
	Retiring a single page from use will have negligible performance impact.
	This is an internal fault;  a repair action will be indicated by an accompanying dimm fault diagnosis.
EOM

create_fault <<-EOM
	$GMEMFLTCLASS.dimm_sb
	A memory module is experiencing excessive correctable errors affecting large numbers of pages.
	A memory module is experiencing excessive correctable errors affecting large numbers of pages.
	Major
	A memory module is experiencing excessive correctable errors affecting large numbers of pages.
	Affected pages will be retired up to a limit.
	Page retirements are capped at a small fraction of memory to avoid performance impact.
	Schedule a repair action to replace the memory module indicated by 'fmadm faulty'.  While the errors are correctable in nature they may be a precursor to an uncorrectable error which will result in downtime.
EOM

create_fault <<-EOM
	$GMEMFLTCLASS.dimm_ue
	A memory module has suffered an uncorrectable error.
	A memory module has suffered an uncorrectable error.
	Major
	A memory module has suffered an uncorrectable error.
	The affected page will be retired from further use.
	Retiring a single page from use will have negligible performance impact.
	Schedule an urgent repair action to replace the memory module indicated by 'fmadm faulty'.
EOM

#
# Modify existing AMD ereport payload
#

#
# Add payload version 1 to ereport.cpu.amd.{dc,ic,bu,ls,nb} for others to
# inherit.  We add them at this subclass level because there is also
# subclass ereport.cpu.amd.mc which has non-MCA ereports.
#
for subclass in dc ic bu ls nb
do
	erevent -P 1=- $AMDERPTSUBCLASS.${subclass} <<-EOM
	IA32_MCG_STATUS:uint64_t:MCG_STATUS value when MCA banks were read:
	machine_check_in_progress:boolean value:MCG_STATUS.MCIP:
	ip:uint64_t:Instruction pointer for machine checks, if MCG_STATUS.EIPV:
	privileged:boolean value:For #MC indicates if interrupted IP was kernel/user:
	bank_number:uint8_t:MCA bank number - N'th set of MCi{CTL,STATUS,ADDR,MISC}:
	bank_msr_offset:uint64_t:MSR offset of MCi_CTL register for this bank:
	IA32_MCi_STATUS:uint64_t:MCi_STATUS raw value:
	overflow:boolean_value:Set if MCi_STATUS.OVER indicates an overflow:
	error_uncorrected:boolean value:Set if MCi_STATUS.UC indicates uncorrected:
	error_enabled:boolean value:Set if MCi_STATUS.EN indicates #MC was enabled for this error type:
	processor_context_corrupt:boolean value:Set if MCi_STATUS.PCC indicates corruption:
	threshold_based_error_status:string:No tracking, Green - below threshold, Yellow - above threshold, Reserved:
	error_code:uint16_t:MCi_STATUS architectural error code - bits 15-0:
	model_specific_error_code:uint16_t:MCi_STATUS model-specific error code - bits 31-16:
	IA32_MCi_ADDR:uint64_t:MCi_ADDR raw value, included if MCi_STATUS.ADDRV indicates address valid:
	IA32_MCi_MISC:uint64_t:MCi_MISC raw value, included if MCi_STATUS.MISCV indicates misc register valid:
EOM
done

# AMD ereports that just inherit generic payload.
# 
# Create a version 1 payload an indicate the correct
# inheritance.

	>ereport:0
	>ereport.cpu:0
	>ereport.cpu.amd:0
	IA32_MCG_STATUS:uint64_t:MCG_STATUS value when MCA banks were read:
	machine_check_in_progress:boolean value:MCG_STATUS.MCIP:
	ip:uint64_t:Instruction pointer for machine checks, if MCG_STATUS.EIPV:
	privileged:boolean value:For #MC indicates if interrupted IP was kernel/user:
	bank_number:uint8_t:MCA bank number - N'th set of MCi{CTL,STATUS,ADDR,MISC}:
	bank_msr_offset:uint64_t:MSR offset of MCi_CTL register for this bank:
	IA32_MCi_STATUS:uint64_t:MCi_STATUS raw value:
	overflow:boolean_value:Set if MCi_STATUS.OVER indicates an overflow:
	error_uncorrected:boolean value:Set if MCi_STATUS.UC indicates uncorrected:
	error_enabled:boolean value:Set if MCi_STATUS.EN indicates #MC was enabled for this error type:
	processor_context_corrupt:boolean value:Set if MCi_STATUS.PCC indicates corruption:
	threshold_based_error_status:string:No tracking, Green - below threshold, Yellow - above threshold, Reserved:
	error_code:uint16_t:MCi_STATUS architectural error code - bits 15-0:
	model_specific_error_code:uint16_t:MCi_STATUS model-specific error code - bits 31-16:
	IA32_MCi_ADDR:uint64_t:MCi_ADDR raw value, included if MCi_STATUS.ADDRV indicates address valid:
	IA32_MCi_MISC:uint64_t:MCi_MISC raw value, included if MCi_STATUS.MISCV indicates misc register valid:
EOM

for ev in "dc.tag_par" "dc.stag_par" "dc.l1tlb_par" "dc.l2tlb_par" \
    "ic.inf_sys_ecc1" "ic.inf_sys_eccm" "ic.inf_l2_ecc1" "ic.inf_l2_eccm" \
    "ic.data_par" "ic.tag_par" "ic.stag_par" "ic.l1tlb_par" "ic.l2tlb_par" \
    "ic.rdde" "bu.l2t_par" "bu.s_rde" "ls.s_rde" "nb.ht_crc" "nb.ht_sync" \
    "nb.gart_walk" "nb.rmw" "nb.wdog" "nb.dramaddr_par"
do
	subclass=$(echo $ev | cut -d. -f1)
	erevent -P 1=- $AMDERPTSUBCLASS.${ev} <<-EOM
	>ereport:0
	>ereport.cpu:0
	>ereport.cpu.amd:0
	>ereport.cpu.amd.${subclass}:1
EOM
done

# AMD ereports that add a syndrome and syndrome type
for ev in "dc.inf_l2_ecc1" "dc.inf_l2_eccm" "dc.data_ecc1" \
    "dc.data_ecc1_uc" "dc.data_eccm" "bu.l2d_ecc1" \
    "bu.l2d_eccm" "bu.l2t_ecc1" "bu.l2t_eccm"
do
	subclass=$(echo $ev | cut -d. -f1)
	erevent -P 1=- $AMDERPTSUBCLASS.${ev} <<-EOM
	>ereport:0
	>ereport.cpu:0
	>ereport.cpu.amd:0
	>ereport.cpu.amd.${subclass}:1
	syndrome:uint64_t:ECC syndrome:
	syndrome-type:string:ECC syndrome type - 'E' for 64/8, 'C' for 128/16:
EOM
done

# AMD ereports that add syndrome, syndrome type, resource
for ev in "dc.inf_sys_ecc1" "dc.inf_sys_eccm" "bu.s_ecc1" \
    "bu.s_eccm" "nb.mem_ce" "nb.mem_ue"
do
	subclass=$(echo $ev | cut -d. -f1)
	erevent -P 1=- $AMDERPTSUBCLASS.${ev} <<-EOM
	>ereport:0
	>ereport.cpu:0
	>ereport.cpu.amd:0
	>ereport.cpu.amd.${subclass}:1
	syndrome:uint64_t:ECC syndrome:
	syndrome-type:string:ECC syndrome type - 'E' for 64/8, 'C' for 128/16:
	resource:fmri array:Array of affected resources:
EOM
done

# AMD ereports that add a stack trace
for ev in "nb.ma" "nb.ta"
do
	subclass=$(echo $ev | cut -d. -f1)
	erevent -P 1=- $AMDERPTSUBCLASS.${ev} <<-EOM
	>ereport:0
	>ereport.cpu:0
	>ereport.cpu.amd:0
	>ereport.cpu.amd.${subclass}:1
	stack:string array:Stack backtrace - optional debug:
EOM
done

exit 0
